     46 (defun rc-clipboard-yank ()
     47   "Copies the active region to the system clipboard."
     48   (interactive)
     49   (when (region-active-p)
     50     (gui-set-selection 'CLIPBOARD
     51       (buffer-substring (region-beginning) (region-end)))))
     52 (defun rc-clipboard-paste ()
     53   "Pastes text from the system clipboard."
     54   (interactive)
     55   (let ((text (gui-get-selection 'CLIPBOARD)))
     56     (when text (insert-for-yank text))))
