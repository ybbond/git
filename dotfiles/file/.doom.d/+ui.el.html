     41 ;; Enable ⌘+c and ⌘+v for clipboard handling
     42 (defun rc-clipboard-yank ()
     43   "Copies the active region to the system clipboard."
     44   (interactive)
     45   (when (region-active-p)
     46     (gui-set-selection 'CLIPBOARD
     47       (buffer-substring (region-beginning) (region-end)))))
     48 (defun rc-clipboard-paste ()
     49   "Pastes text from the system clipboard."
     50   (interactive)
     51   (let ((text (gui-get-selection 'CLIPBOARD)))
     52     (when text (insert-for-yank text))))
     53 (map! :v "s-c" #'rc-clipboard-yank
     54       :nvi "s-v" #'rc-clipboard-paste)
     55 (define-key! :keymaps '(evil-ex-completion-map) "s-v" #'rc-clipboard-paste)
