    349 function ToggleNumberToggle(numberVar)
    350   " Reset group
    351   augroup numbertoggle
    352     autocmd!
    353   augroup end
    354 
    355   " Enable if toggled on
    356   if a:numberVar
    357     augroup numbertoggle
    358       if &number == 1
    359         autocmd!
    360         autocmd BufEnter,FocusGained,InsertLeave * set relativenumber
    361         autocmd BufLeave,FocusLost,InsertEnter * set norelativenumber
    362       endif
    363     augroup end
    364   endif
    365 endfunction

    367 function! s:goyo_enter()
    368   if executable('tmux') && strlen($TMUX)
    369     silent !tmux set status off
    370     silent !tmux list-panes -F '\#F' | grep -q Z || tmux resize-pane -Z
    371   endif
    372   set noshowcmd
    373   set wrap
    374   set nonumber
    375   set norelativenumber
    376   set eventignore=FocusGained,BufEnter
    377   set nocursorline
    378   Limelight
    379   :call ToggleNumberToggle(0)
    380 endfunction

    382 function! s:goyo_leave()
    383   if executable('tmux') && strlen($TMUX)
    384     silent !tmux set status on
    385     silent !tmux list-panes -F '\#F' | grep -q Z && tmux resize-pane -Z
    386   endif
    387   set showcmd
    388   set nowrap
    389   set number
    390   set relativenumber
    391   set eventignore=
    392   set cursorline
    393   Limelight!
    394   :call ToggleNumberToggle(1)
    395 endfunction
