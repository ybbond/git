     24 function makin(input) {
     25   R = hexToR(input);
     26   G = hexToG(input);
     27   B = hexToB(input);
     28 
     29   return `rgb(${R.toString()}, ${G.toString()}, ${B.toString()})`;

     32 function hexToR(h) {
     33   return parseInt(cutHex(h).substring(0, 2), 16);
     34 }
     35 function hexToG(h) {
     36   return parseInt(cutHex(h).substring(2, 4), 16);
     37 }
     38 function hexToB(h) {
     39   return parseInt(cutHex(h).substring(4, 6), 16);

     41 function cutHex(h) {
     42   return h.charAt(0) == "#" ? h.substring(1, 7) : h;
