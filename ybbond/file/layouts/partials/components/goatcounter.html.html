     61 	// Object to urlencoded string, starting with a ?.
     62 	var urlencode = function(obj) {
     63 		var p = []
     64 		for (var k in obj)
     65 			if (obj[k] !== '' && obj[k] !== null && obj[k] !== undefined && obj[k] !== false)
     66 				p.push(encodeURIComponent(k) + '=' + encodeURIComponent(obj[k]))
     67 		return '?' + p.join('&')
